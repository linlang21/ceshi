-- 2. RECOVER (REPLACED WITH USER-SUPPLIED IMPLEMENTATION)
_G.RunRecover = function()
    local function main()
        local mp_local = G and G.main_player
        if not mp_local then
            print("[Recover] ERROR: Main player not found")
            return
        end
        local mod = "hexm.client.ui.windows.gm.gm_combat.combat_train_action"
        local action = nil
        pcall(function() action = portable.import(mod) end)
        if not action then
            print("[Recover] Module "..mod.." not found, using buff fallback")
            -- Fallback: apply recovery buff directly
            pcall(mp_local.add_buff, mp_local, 70141) -- recover buff
            pcall(mp_local.add_buff, mp_local, 70063) -- optional: stamina/full recovery buff if needed
        else
            -- Preferred: use combat action API
            pcall(function() action.recover_hp(1) end) -- recover HP
            pcall(function() action.fullfill_all_combat_res(1) end) -- fill stamina
            print("[Recover] Recovery applied via combat action")
        end
    end
    pcall(main)
end
-- 3. KILL NPC
_G.RunKillNPC = function()
    write_debug("[KillNPC] Triggered")
    local count = 0
    local combat_action = nil
    pcall(function() combat_action = portable.import('hexm.client.ui.windows.gm.gm_combat.combat_train_action') end)
    if combat_action then
        pcall(function() combat_action.set_npc_mortal(true) end)
        pcall(function() combat_action.kill_all_npc() end)
    end
    -- Boss Damage Logic
    pcall(function()
        local target_id = mp:get_lock_target_id()
        if target_id then
            local target = G.space:get_entity(target_id)
            if target then
                pcall(function() target:force_set_HP(0, mp.entity_id, "gm") end)
                pcall(function() target:attr_set_HP(0, mp.entity_id, true, false) end)
                pcall(function() target:do_direct_damage(999999999, mp.entity_id, 0, 0, 0, 0) end)
                count = count + 1
            end
        end
    end)
    -- Area Cleanup Loop
    pcall(function()
        local all = MEntityManager:GetAOIEntities()
        for i = 1, #all do
            local ent = all[i]
            local ok, name = pcall(function() return ent:GetName() end)
            if ok and name and (name:find('AiAvatar') or name:find('Npc') or name:find('Boss')) then
                local ok2, eid = pcall(function() return ent.entity_id end)
                if ok2 and eid then
                    local target = G.space:get_entity(eid)
                    if target and target ~= mp then
                        pcall(function() target:force_set_HP(0, mp.entity_id, "gm") end)
                        pcall(function() target:do_direct_damage(999999999, mp.entity_id, 0, 0, 0, 0) end)
                        count = count + 1
                    end
                end
            end
        end
    end)
end
-- 4. AUTO LOOT
-- === Robust AutoLoot with interactable filter ===
_G.RunAutoLoot = function()
    local mp_local = G and G.main_player
    if not mp_local then
        write_debug("[AutoLoot] ERROR: main player not found")
        return
    end
    local interact_misc = nil
    pcall(function() interact_misc = portable.import('hexm.common.misc.interact_misc') end)
    -- 1) Collections
    pcall(function() mp_local:ride_skill_collect_nearby_collections(5000) end)
    -- 2) Rewards
    pcall(function()
        local rewards = mp_local:ride_skill_find_nearest_kill_reward(5000)
        if rewards then
            pcall(function() mp_local:ride_skill_get_kill_reward(rewards) end)
        end
    end)
    -- 3) Drops
    pcall(function()
        local drops = DropManager.get_nearby_drop_entities(5000)
        if drops then
            for _, eid in ipairs(drops) do
                pcall(function() mp_local:pick_drop_item(eid) end)
                pcall(function() mp_local:pick_reward_item(eid) end)
            end
        end
    end)
    -- 4) Interact entities (complex interactions) with filtering
    local ok_pos, playerPos = pcall(function() return mp_local:get_position() end)
    if not ok_pos or not playerPos then playerPos = {x=0,y=0,z=0} end
    local entities = MEntityManager:GetAOIEntities() or {}
    local targets = {}
    -- Name patterns to ignore (case-insensitive)
    local IGNORE_NAME_PATTERNS = {
        "lockpick", "lock pick", "elevator", "elev", "mini", "minigame", "mini-game",
        "puzzle", "chess", "rhythm", "pitch", "pitch pot", "slot", "gacha"
    }
    -- Status/config IDs to ignore (populate with IDs you want to blacklist)
    local IGNORE_STATUS_IDS = {
        -- [12345] = true, -- example: add known status_no values to skip
    }
    local IGNORE_CONFIG_IDS = {
        -- [54321] = true, -- example: add known config_no values to skip
    }
    local function name_matches_ignore(name)
        if not name then return false end
        local lname = string.lower(name)
        for _, pat in ipairs(IGNORE_NAME_PATTERNS) do
            if lname:find(pat, 1, true) then
                return true
            end
        end
        return false
    end
    for i = 1, #entities do
        local ent = entities[i]
        local ok, name = pcall(function() return ent:GetName() end)
        if ok and name and name:find('InteractComEntity') then
            -- Quick name-based filter
            if name_matches_ignore(name) then
                write_debug("[AutoLoot] Skipping by name: " .. tostring(name))
            else
                local ok2, eno = pcall(function() return ent:GetEntityNo() end)
                local ok3, eid = pcall(function() return ent.entity_id end)
                if ok2 and ok3 and eid then
                    local luaEnt = G.space and G.space:get_entity(eid)
                    if luaEnt then
                        local ok4, comp = pcall(function() return luaEnt:get_interact_comp(eid) end)
                        if ok4 and comp and comp.position then
                            -- Component-based filter: check status/config IDs
                            local skip = false
                            -- check status_no (could be single or table)
                            if comp.status_no then
                                if type(comp.status_no) == "table" then
                                    for _, sid in ipairs(comp.status_no) do
                                        if IGNORE_STATUS_IDS[sid] then skip = true; break end
                                    end
                                else
                                    if IGNORE_STATUS_IDS[comp.status_no] then skip = true end
                                end
                            end
                            -- check config_no (could be single or table)
                            if not skip and comp.config_no then
                                if type(comp.config_no) == "table" then
                                    for _, cid in ipairs(comp.config_no) do
                                        if IGNORE_CONFIG_IDS[cid] then skip = true; break end
                                    end
                                else
                                    if IGNORE_CONFIG_IDS[comp.config_no] then skip = true end
                                end
                            end
                            if skip then
                                write_debug(string.format("[AutoLoot] Skipping entity id=%s due to blacklisted comp id", tostring(eid)))
                            else
                                local dx = playerPos.x - comp.position[1]
                                local dy = playerPos.y - comp.position[2]
                                local dz = playerPos.z - comp.position[3]
                                local dist = math.sqrt(dx*dx + dy*dy + dz*dz)
                                local priority = tostring(eid):find('ins_entity') and 0 or 1
                                table.insert(targets, {
                                    entity_no = eno,
                                    entity_id = eid,
                                    luaEnt = luaEnt,
                                    comp = comp,
                                    distance = dist,
                                    priority = priority
                                })
                            end
                        end
                    end
                end
            end
        end
    end
    table.sort(targets, function(a, b)
        if a.priority ~= b.priority then return a.priority < b.priority end
        return a.distance < b.distance
    end)
    for i = 1, #targets do
        local t = targets[i]
        local ways = {}
        local seen = {}
        if interact_misc then
            local ok_ways, possible = pcall(function()
                return interact_misc.get_all_possible_active_ways(t.entity_no)
            end)
            if ok_ways and possible then
                for _, w in ipairs(possible) do
                    if not seen[w] then seen[w] = true; table.insert(ways, w) end
                end
            end
        end
        local comp_id = nil
        if t.comp and t.comp.components then
            for cid, comp_data in pairs(t.comp.components) do
                comp_id = cid
                if comp_data.status_no and not seen[comp_data.status_no] then
                    seen[comp_data.status_no] = true
                    table.insert(ways, comp_data.status_no)
                end
                if comp_data.config_no and not seen[comp_data.config_no] then
                    seen[comp_data.config_no] = true
                    table.insert(ways, comp_data.config_no)
                end
            end
        end
        if #ways > 0 then
            pcall(function() mp_local:set_interact_target_id(t.entity_id) end)
            for _, way in ipairs(ways) do
                pcall(function()
                    mp_local:trigger_active_interact(way, t.entity_id, nil, nil, comp_id)
                end)
            end
            pcall(function() mp_local:trigger_active_interact() end)
        end
    end
    write_debug("[AutoLoot] Completed run")
end

local function apply_new_gm_speed(speed)
    local mod = "hexm.client.ui.windows.gm.gm_combat.combat_train_action"
    local action = nil
    pcall(function() action = portable.import(mod) end)
    if action and action.set_game_speed then

    end

end
local function apply_old_dialog_speed(speed, enable)
    if G then
        if G.dialog_global_time_scale ~= nil then G.dialog_global_time_scale = speed end
        if G.space then
            local space = G.space
            if space.dialog_global_time_scale ~= nil then space.dialog_global_time_scale = speed end
            if space.dialog_set_global_time_scale then pcall(space.dialog_set_global_time_scale, space, speed) end
            if space.imp_dialogs_manager then
                local dm = space.imp_dialogs_manager
                if dm.dialog_set_global_time_scale then pcall(dm.dialog_set_global_time_scale, dm, speed) end
            end
        end
        local mp = G.main_player
        if mp and mp.dialog_set_time_speed_scale then
            pcall(mp.dialog_set_time_speed_scale, mp, enable, speed)
        end
    end
end
_G.RunSetSpeed = function(valIndex)
    local speeds = {1.0, 1.5, 3.0, 5.0, 10.0, 30.0}
    local targetSpeed = speeds[valIndex] or 1.0
    local isEnable = (valIndex > 1)
    apply_new_gm_speed(targetSpeed)
    apply_old_dialog_speed(targetSpeed, isEnable)
    write_debug("Set Speed to: x" .. targetSpeed)
end

-- 9. Remove Buffs
_G.RunRemoveBuffs = function()
    write_debug("[RemoveBuffs] Starting buff removal")
    local mp_local = G and G.main_player
    if not mp_local then
        write_debug("[RemoveBuffs] ERROR: main player not found")
        return
    end
    local mod_name = "hexm.client.ui.windows.gm.gm_combat.combat_train_action"
    local action = nil
    pcall(function() action = portable.import(mod_name) end)
    write_debug("[RemoveBuffs] combat_train_action loaded: " .. tostring(action ~= nil))
    local function safe_call(desc, fn)
        local ok, err = pcall(fn)
        if ok then
            write_debug("[RemoveBuffs] ✔ " .. desc)
            return true
        else
            write_debug("[RemoveBuffs] ✘ " .. desc .. " -> " .. tostring(err))
            return false
        end
    end
    -- Try multiple ways to detect whether the player currently has a buff.
    local function has_buff(buff_id)
        local ok, res
        ok, res = pcall(function()
            if mp_local.has_buff then return mp_local.has_buff(mp_local, buff_id) end
            if mp_local["has_buff"] and type(mp_local["has_buff"]) == "function" then return mp_local:has_buff(buff_id) end
            return nil
        end)
        if ok and res ~= nil then return res end
        ok, res = pcall(function()
            if mp_local.get_buff then return mp_local.get_buff(mp_local, buff_id) ~= nil end
            if mp_local["get_buff"] and type(mp_local["get_buff"]) == "function" then return mp_local:get_buff(buff_id) ~= nil end
            return nil
        end)
        if ok and res ~= nil then return res end
        ok, res = pcall(function()
            if mp_local.buffs and type(mp_local.buffs) == "table" then
                if mp_local.buffs[buff_id] then return true end
                for _, v in pairs(mp_local.buffs) do
                    if type(v) == "table" and (v.id == buff_id or v.buff_id == buff_id) then
                        return true
                    end
                end
            end
            return nil
        end)
        if ok and res ~= nil then return res end
        return false
    end
    local function remove_buff(buff_id)
        local eid = mp_local and mp_local.entity_id
        local removed = false
        if action and action.rm_buff then
            removed = safe_call(string.format("action.rm_buff(%d)", buff_id), function() action.rm_buff(buff_id) end) or removed
            if removed then return true end
            removed = safe_call(string.format("action:rm_buff(%d)", buff_id), function() action:rm_buff(buff_id) end) or removed
            if removed then return true end
        end
        if action and action.remove_buff then
            removed = safe_call(string.format("action.remove_buff(%d, %s)", buff_id, tostring(eid)), function() action.remove_buff(buff_id, eid) end) or removed
            if removed then return true end

-- Function to create buff toggle
local function makeBuffToggle(buffId, buffName, varName)
    _G[varName] = _G[varName] or false
   
    local toggleFunc = function()
        _G[varName] = not _G[varName]
        if _G[varName] then
            pcall(mp.add_buff, mp, buffId)
            write_debug("[Buff] Added: " .. buffName .. " (ID: " .. buffId .. ")")
        else
            local a = portable.import("hexm.client.ui.windows.gm.gm_combat.combat_train_action")
            if a then a.rm_buff(buffId) end
            write_debug("[Buff] Removed: " .. buffName .. " (ID: " .. buffId .. ")")
        end
    end
   
    return toggleFunc, function() return _G[varName] end,
        function() return buffName .. " < " .. (_G[varName] and "ON" or "OFF") .. " >" end
end

-- Function to create Buff List Menu
function createBuffListMenu()
    -- Remove old buff menu if exists
    if _G.BUFF_MENU then
        _G.BUFF_MENU:removeFromParent()
        _G.BUFF_MENU = nil
    end
   
    write_debug("[Buff Menu] Creating Buff List Menu")
   
    -- EXACT RGB COLORS (same as main menu)
    local C_BG = cc.c3b(20, 20, 25) -- Dark Background
    local C_TEXT_OFF = cc.c3b(192, 192, 192) -- Inactive (Silver)
    local C_TEXT_ON = cc.c3b(0, 255, 0) -- Active (Green)
    local C_SELECTED = cc.c3b(255, 0, 255) -- Hover/Selected (Magenta)
    local C_READONLY = cc.c3b(153, 255, 255) -- Read-only/value part (Cyan)
   
    -- DIMENSIONS (same as main menu)
    local btnHeight = 90
    local spacing = 16
    local headerSpace = 160
    local footerSpace = 400
    local panelWidth = 940
    local buttonWidth = panelWidth - 60
   
        -- Calculate full content height (ALL buttons, even if screen is smaller)
    local totalButtons = #buffList
    local fullContentHeight = totalButtons * (btnHeight + spacing)
 
    -- Bottom padding inside panel (so content doesn't touch the bottom)
    local listBottomPadding = 40
 
    -- Clamp panel height to screen, but KEEP fullContentHeight for scrolling
    local maxHeight = size.height - 100
    local totalHeight = math.min(headerSpace + fullContentHeight + listBottomPadding, maxHeight)
 
    -- Visible list window (area where buttons are allowed to be visible)
    local listViewportH = totalHeight - headerSpace - listBottomPadding
    if listViewportH < btnHeight then listViewportH = btnHeight end
 
   
    -- Create panel
    local panel = ccui.Layout:create()
    panel:setContentSize(cc.size(panelWidth, totalHeight))
    panel:setBackGroundColorType(1)
    panel:setBackGroundColor(C_BG)
    panel:setBackGroundColorOpacity(240)
   
    -- Position at center
    local startX = (size.width - panelWidth) / 2
    local startY = (size.height - totalHeight) / 2
    panel:setPosition(cc.p(startX, startY))
   
    scene:addChild(panel, 9999)
    _G.BUFF_MENU = panel
 
    -- Create drag area (header) to handle dragging without swallowing all touches
    local dragAreaHeight = 100
    local dragArea = ccui.Layout:create()
    dragArea:setContentSize(cc.size(panelWidth, dragAreaHeight))
    dragArea:setPosition(cc.p(0, totalHeight - dragAreaHeight))
    panel:addChild(dragArea)
   
    -- Header
    local header = ccui.Text:create("Buff List Menu", "Arial", 54)
    header:setColor(cc.c3b(255, 230, 120))
    header:enableOutline(cc.c4b(0, 255, 0, 255), 2)
    header:setAnchorPoint(cc.p(0.5, 1))
    header:setPosition(cc.p(panelWidth / 2, dragAreaHeight - 40))
    dragArea:addChild(header)
   
    -- Create buff toggle buttons
    local buffButtons = {}
    local buffSelectedIndex = 1
	
	    -- ==========================
    -- SCROLLABLE / CLIPPED LIST
    -- ==========================
    local listViewport = ccui.Layout:create()
    listViewport:setContentSize(cc.size(panelWidth, listViewportH))
    listViewport:setPosition(cc.p(0, listBottomPadding))
    listViewport:setClippingEnabled(true)
    panel:addChild(listViewport)
 
    local listContent = ccui.Layout:create()
    listContent:setAnchorPoint(cc.p(0, 1)) -- top-left
    listContent:setContentSize(cc.size(panelWidth, fullContentHeight))
    listContent:setPosition(cc.p(0, listViewportH)) -- align top of content to top of viewport
    listViewport:addChild(listContent)
 
    local scroll = 0
    local maxScroll = math.max(0, fullContentHeight - listViewportH)
 
    local function applyScroll(newScroll)
        scroll = math.max(0, math.min(maxScroll, newScroll))
        -- with anchor top, pushing position UP reveals lower items
        listContent:setPosition(cc.p(0, listViewportH + scroll))
    end
 
    local function ensureSelectedVisible()
        if maxScroll <= 0 then return end
 
        local idx = buffSelectedIndex
        local btnTop = fullContentHeight - ((idx - 1) * (btnHeight + spacing))
        local btnBottom = btnTop - btnHeight
 
        local yMax = fullContentHeight - scroll
        local yMin = yMax - listViewportH
 
        -- selected is ABOVE visible window -> scroll up (toward top)
        if btnTop > yMax then
            applyScroll(fullContentHeight - btnTop)
        -- selected is BELOW visible window -> scroll down
        elseif btnBottom < yMin then
            applyScroll(fullContentHeight - listViewportH - btnBottom)
        end
    end
 
   
    for i, buff in ipairs(buffList) do
        local varName = "BUFF_" .. buff.id
        local execFunc, isActiveFunc, dynamicLabelFunc = makeBuffToggle(buff.id, buff.name, varName)
       
        local b = ccui.Button:create()
        b:setTitleFontSize(56)
        b:setScale9Enabled(true)
        b:setContentSize(cc.size(buttonWidth, btnHeight))
        b:setColor(C_BG)
        b:setTitleText("")
       
        b:setAnchorPoint(cc.p(0.5, 1))
        b:setPosition(cc.p(panelWidth / 2, fullContentHeight - ((i-1) * (btnHeight + spacing))))
       
        -- Text node for the button
        local mainTextNode = ccui.Text:create("", "Arial", 56)
        mainTextNode:setAnchorPoint(cc.p(0.5, 0.5))
        mainTextNode:setPosition(cc.p(buttonWidth / 2, btnHeight / 2))
        b:addChild(mainTextNode, 1)
       
        local btnObj = {
            btn = b,
            mainText = mainTextNode,
            execFunc = execFunc,
            isActive = isActiveFunc,
            updateText = function()
                local text = buff.name
                if dynamicLabelFunc then
                    text = dynamicLabelFunc()
                elseif isActiveFunc then
                    local val = isActiveFunc()
                    text = buff.name .. (val and " < ON >" or " < OFF >")
                end
                mainTextNode:setString(text)
               
                -- Update color based on state
                local isSelected = (i == buffSelectedIndex)
                local targetColor = C_TEXT_OFF
               
                if isActiveFunc and isActiveFunc() then
                    targetColor = C_TEXT_ON
                end
               
                if isSelected then
                    targetColor = C_SELECTED
                    b:setScale(1.05)
                else
                    b:setScale(1.0)
                end
               
                mainTextNode:setColor(targetColor)
            end
        }
       
        b:addTouchEventListener(function(sender, eventType)
            if eventType == 2 then
                if execFunc then pcall(execFunc) end
                btnObj.updateText()
            end
        end)
       
        listContent:addChild(b)
        table.insert(buffButtons, btnObj)
    end
   
    -- Function to update buff button visuals
    local function updateBuffSelection()
        for i, btnObj in ipairs(buffButtons) do
            local isSelected = (i == buffSelectedIndex)
            local targetColor = C_TEXT_OFF
           
            if btnObj.isActive and btnObj.isActive() then
                targetColor = C_TEXT_ON
            end
           
            if isSelected then
                targetColor = C_SELECTED
                btnObj.btn:setScale(1.05)
            else
                btnObj.btn:setScale(1.0)
            end
           
            btnObj.mainText:setColor(targetColor)
        end
		ensureSelectedVisible()
 
    end
   
    -- Initialize button texts
    for _, btnObj in ipairs(buffButtons) do
        if btnObj.updateText then btnObj.updateText() end
    end
   
    updateBuffSelection()
	applyScroll(0)
 
   
    -- Navigation functions for buff menu
    _G.BuffMenuUp = function()
        buffSelectedIndex = buffSelectedIndex - 1
        if buffSelectedIndex < 1 then buffSelectedIndex = #buffButtons end
        updateBuffSelection()
    end
   
    _G.BuffMenuDown = function()
        buffSelectedIndex = buffSelectedIndex + 1
        if buffSelectedIndex > #buffButtons then buffSelectedIndex = 1 end
        updateBuffSelection()
    end
   
    _G.BuffMenuConfirm = function()
        local btnObj = buffButtons[buffSelectedIndex]
        if btnObj and btnObj.execFunc then
            pcall(btnObj.execFunc)
            btnObj.updateText()
        end
    end
   
    _G.BuffMenuBack = function()
        if _G.CloseBuffMenu then pcall(_G.CloseBuffMenu) end
    end
   
    -- Close Button
    local closeBtn = ccui.Button:create()
    closeBtn:setTitleText("X")
    closeBtn:setTitleFontSize(48)
    closeBtn:setTitleColor(cc.c3b(255, 80, 80))
    closeBtn:setPosition(cc.p(panelWidth - 50, dragAreaHeight - 50))
    closeBtn:addTouchEventListener(function(s,e)
        if e == 2 then
            _G.BuffMenuBack()
        end
    end)
    dragArea:addChild(closeBtn)
   
    -- Minimize Button
    local btn_minimize = ccui.Button:create()
    btn_minimize:setTitleText("−")
    btn_minimize:setTitleFontSize(52)
    btn_minimize:setTitleColor(cc.c3b(255, 255, 120))
    btn_minimize:setPosition(cc.p(panelWidth - 130, dragAreaHeight - 50))
    btn_minimize:addTouchEventListener(function(s,e)
        if e == 2 then
            panel:setVisible(false)
            setFocusToMain()  -- Restore main menu focus on minimize/hide
        end
    end)
    dragArea:addChild(btn_minimize)
   
    -- Dragging Logic (now on dragArea instead of panel)
    dragArea:setTouchEnabled(true)
    local isDragging = false
    local dragOffset = nil
   
    local function keepPanelInBounds(newX, newY)
        local pSize = panel:getContentSize()
        local minX, maxX = 0, size.width - pSize.width
        local minY, maxY = 0, size.height - pSize.height
        return math.max(minX, math.min(maxX, newX)), math.max(minY, math.min(maxY, newY))
    end
   
    dragArea:addTouchEventListener(function(sender, eventType)
        if eventType == 0 then
            local tPos = sender:getTouchBeganPosition()
            local pPos = panel:getPosition()
            dragOffset = cc.p(tPos.x - pPos.x, tPos.y - pPos.y)
            isDragging = true
            return true
        elseif eventType == 1 and isDragging then
            local tPos = sender:getTouchMovePosition()
            local newX, newY = keepPanelInBounds(tPos.x - dragOffset.x, tPos.y - dragOffset.y)
            panel:setPosition(cc.p(newX, newY))
            return true
        elseif eventType == 2 or eventType == 3 then
            isDragging = false
            return true
        end
    end)
   
    write_debug("[Buff Menu] Created with " .. #buffButtons .. " buffs")
    return panel
end
-- Function to toggle buff menu visibility
_G.ToggleBuffMenu = function()
    if not _G.BUFF_MENU or not _G.BUFF_MENU:getParent() then
        createBuffListMenu()
        _G.BUFF_MENU:setVisible(true)
        setFocusToBuff()
		_G.BuffMenuOpen = true
        write_debug("[Buff Menu] Opened")
    else
        local isVisible = _G.BUFF_MENU:isVisible()
        _G.BUFF_MENU:setVisible(not isVisible)
        if not isVisible then
            setFocusToBuff()
        else
            setFocusToMain()
        end
        write_debug("[Buff Menu] Toggled visibility: " .. tostring(_G.BUFF_MENU:isVisible()))
		_G.BuffMenuOpen = (_G.BUFF_MENU and _G.BUFF_MENU:getParent()) and true or false
    end
end

local function HandleLoopAction(state, actionRef, func, delaySec)
    -- default delay if nil/invalid
    delaySec = tonumber(delaySec) or 5.0
    if delaySec < 0.05 then delaySec = 0.05 end
    -- stop old loop action
    if actionRef then
        local scene = cc.Director:getInstance():getRunningScene()
        if scene then scene:stopAction(actionRef) end
        actionRef = nil
    end
    -- EXECUTE IMMEDIATELY regardless of Once or Loop
    if state == 1 or state == 2 then
        func()
    end
    -- IF LOOP: Start the timer for the *next* execution
    if state == 2 then
        local scene = cc.Director:getInstance():getRunningScene()
        if scene then
            actionRef = cc.RepeatForever:create(cc.Sequence:create({
                cc.DelayTime:create(delaySec),
                cc.CallFunc:create(function()
                    func()
                end)
            }))
            scene:runAction(actionRef)
        end
    end
    return actionRef
end
_G.ToggleAutoLootLoop = function()
    _G.LOOT_STATE = (_G.LOOT_STATE + 1) % 3
    loot_action = HandleLoopAction(_G.LOOT_STATE, loot_action, _G.RunAutoLoot, _G.LOOT_LOOP_SEC)

_G.CycleSpeed = function(dir)
    _G.SPEED_INDEX = _G.SPEED_INDEX + dir
    if _G.SPEED_INDEX < 1 then _G.SPEED_INDEX = #SPEED_LABELS end
    if _G.SPEED_INDEX > #SPEED_LABELS then _G.SPEED_INDEX = 1 end
    _G.RunSetSpeed(_G.SPEED_INDEX)
    UpdateButtonVisuals()
end
_G.CycleState = function(globalVarName, dir, maxState)
    local val = _G[globalVarName] + dir
    if val < 0 then val = maxState end
    if val > maxState then val = 0 end
    _G[globalVarName] = val
    UpdateButtonVisuals()
end

-- 1. EXACT RGB COLORS
local C_BG = cc.c3b(20, 20, 25) -- Dark Background
local C_TEXT_OFF = cc.c3b(192, 192, 192) -- Inactive (Silver)
local C_TEXT_ON = cc.c3b(0, 255, 0) -- Active (Green)
local C_SELECTED = cc.c3b(255, 0, 255) -- Hover/Selected (Magenta)
local C_READONLY = cc.c3b(153, 255, 255) -- Read-only/value part (Cyan)
-- 2. SELECTION LOGIC
local function UpdateSelection()
    for i, item in ipairs(buttonList) do
        local isSelected = (i == selectedIndex)
        -- Default to Inactive Color
        local targetMainColor = C_TEXT_OFF
        -- If the option is Active (ON/LOOP/etc), turn Green
        if item.isActive and item.isActive() then
            targetMainColor = C_TEXT_ON
        end
        -- Selected overrides main color only
        if isSelected then
            targetMainColor = C_SELECTED
            item.btn:setScale(1.05)
        else
            item.btn:setScale(1.0)
        end
        -- We render ALL button text via ccui.Text nodes now
        if item.mainText then
            item.mainText:setColor(targetMainColor)
        end
        -- Value part always cyan (only exists on value buttons)
        if item.valueText then
            item.valueText:setColor(C_READONLY)
        end
    end
end
UpdateButtonVisuals = function()
    for _, item in ipairs(buttonList) do
        if item.updateText then item.updateText() end
    end
    UpdateSelection()
end
-- Navigation Logic
_G.MenuUp = function()
    selectedIndex = selectedIndex - 1
    if selectedIndex < 1 then selectedIndex = #buttonList end
    UpdateSelection()
end
_G.MenuDown = function()
    selectedIndex = selectedIndex + 1
    if selectedIndex > #buttonList then selectedIndex = 1 end
    UpdateSelection()
end
-- Left/Right now changes VALUES (Numpad 4/6)
_G.MenuLeft = function()
    local item = buttonList[selectedIndex]
    if item and item.cycleFunc then
        item.cycleFunc(-1)
        UpdateButtonVisuals()
    end
end
_G.MenuRight = function()
    local item = buttonList[selectedIndex]
    if item and item.cycleFunc then
        item.cycleFunc(1)
        UpdateButtonVisuals()
    end
end
-- Confirm now EXECUTES (Numpad 5)
_G.MenuConfirm = function()
    local item = buttonList[selectedIndex]
    if item and item.execFunc then
        pcall(item.execFunc)
    end
    UpdateButtonVisuals()
end

-- Helper: toggle NPC AI (keeps separate from buff logic)
local function toggle_npc_ai()
    local gm_decorator = package.loaded["hexm.client.debug.gm.gm_decorator"]
        or require("hexm.client.debug.gm.gm_decorator")
    if not gm_decorator.gm_command_short_cuts or not gm_decorator.gm_command_short_cuts.game then
        write_debug("[NPC AI] gm_decorator shortcuts not available")
        return _G.GM_NPCAI
    end
    local cmds = gm_decorator.gm_command_short_cuts.game
    if _G.GM_NPCAI then
        -- Disable NPC AI
        if cmds["$stop_all_npc_ai"] then pcall(cmds["$stop_all_npc_ai"], 1) end
        if cmds["$gm_stop_ai"] then pcall(cmds["$gm_stop_ai"], 1) end
        _G.GM_NPCAI = false
        write_debug("[NPC AI] Disabled")
        return false
    else
        -- Enable NPC AI
        if cmds["$start_all_npc_ai"] then pcall(cmds["$start_all_npc_ai"], 1) end
        if cmds["$gm_start_ai"] then pcall(cmds["$gm_start_ai"], 1) end
        _G.GM_NPCAI = true
        write_debug("[NPC AI] Enabled")
        return true
    end
end

-- Boundary Check
local function keepPanelInBounds(newX, newY)
    local pSize = panel:getContentSize()
    local minX, maxX = 0, size.width - pSize.width
    local minY, maxY = 0, size.height - pSize.height
    return math.max(minX, math.min(maxX, newX)), math.max(minY, math.min(maxY, newY))
end
-- Draggable Logic (now on dragArea)
dragArea:setTouchEnabled(true)
local isDragging = false
local dragOffset = nil
dragArea:addTouchEventListener(function(sender, eventType)
    if eventType == 0 then
        local tPos = sender:getTouchBeganPosition()
        local pPos = panel:getPosition()
        dragOffset = cc.p(tPos.x - pPos.x, tPos.y - pPos.y)
        isDragging = true
        return true
    elseif eventType == 1 and isDragging then
        local tPos = sender:getTouchMovePosition()
        local newX, newY = keepPanelInBounds(tPos.x - dragOffset.x, tPos.y - dragOffset.y)
        panel:setPosition(cc.p(newX, newY))
        return true
    elseif eventType == 2 or eventType == 3 then
        isDragging = false
        return true
    end
end)
